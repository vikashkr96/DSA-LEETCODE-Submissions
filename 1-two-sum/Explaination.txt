class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int>indices ;
        for(int i=0; i<nums.size(); i++){
            for(int j=i+1; j<nums.size() ; j++){
                if (nums[i] + nums[j] == target){
                    indices.push_back(i);
                    indices.push_back(j);
                }
            }
        }
        return indices;
    }
};

> firsly i have created a vector name "indices" for storing indices for final output
> then after i have used a for loop traversing every element throughout the length 
> then after i have used another loop traversing from next element in order to make pair of two numbers to calculate the sum
> then after i've compared the target sum with pair sum of indices i and j
> then after the condition is true i have pushed the indices of both ith and jth element into the vector array that i had created at the first
> then after i have returned the vector array containing indices of that two numbers whose sum is equals to the target sum
